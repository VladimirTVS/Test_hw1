/*
7. Наибольшее из пяти чисел

Ввести пять чисел и вывести наибольшее из них.
Нужно напечатать наибольшее число.

Данные на входе:  Пять целых чисел, разделённых пробелом
Данные на выходе: Одно целое число

Данные на входе:  4 15 9 56 4
Данные на выходе: 56

*/

#include <stdio.h>

int main(void)
{
	int a1, a2, a3, a4, a5;
	scanf("%d%d%d%d%d", &a1, &a2, &a3, &a4, &a5);

	int max = a1;
	max = a2 > max ? a2 : max;
	max = a3 > max ? a3 : max;
	max = a4 > max ? a4 : max;
	max = a5 > max ? a5 : max;

	printf("%d\n", max);

	return 0;
}
